                    iPagina = 0;

                        iPagina = i * 24;

                        if (i > 0) {
                            sbLink.Append("?offset=");
                            sbLink.Append(iPagina);
                        }

                        oDoc = web.Load(sbLink.ToString());

                        foreach (HtmlNode h in hc) 
                        {
                            if (h.InnerText.Contains("window[\"impressions\"]=")) 
                            {
                                sJson = h.InnerText.Replace("window[\"impressions\"]=", string.Empty).Replace("}];", "}]");

                            }
                        }

                        if (sJson == string.Empty) {
                            break;
                        }

                        lProductTemp = Deserialize<List<CarrefourProduct>>(sJson);
